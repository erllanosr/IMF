 internal class persona
    {
        private string name;    
        private int id; 
        private int age;

        public persona(string name, int id, int age)
        {
            this.Name = name;
            this.id = id;
            this.age = age;
        }

        public string Name { get => name; set => name = value; }
        public int Id { get => id; set => id = value; }
        public int Age { get => age; set => age = value; }
    }
