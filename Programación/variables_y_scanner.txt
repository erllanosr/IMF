//INT

import java.util.Scanner;

public class DiaSemana {

	public static void main(String[] args) {
		
		int dato;
		int dato2;
		Scanner sc=new Scanner(System.in);
		
		dato=10;
		System.out.println("Introduce el dato");
		dato2=sc.nextInt();
		
		System.out.println("dato: " + dato + " dato2: " + dato2);
		
	}

}
